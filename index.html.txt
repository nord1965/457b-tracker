// ── LIVE ETF FETCH via Twelve Data (CORS-enabled, free tier, works in browser) ──
const TWELVE_KEY = "7979a2daa7f2481f97a3f7a336334c71";
async function fetchPrice(ticker) {
  const url = `https://api.twelvedata.com/quote?symbol=${ticker}&apikey=${TWELVE_KEY}`;
  const controller = new AbortController();
  const tid = setTimeout(() => controller.abort(), 8000);
  try {
    const res = await fetch(url, { signal: controller.signal });
    clearTimeout(tid);
    if (!res.ok) return null;
    const data = await res.json();
    if (data.status === "error" || data.percent_change === undefined) return null;
    return { changePct: parseFloat(data.percent_change) };
  } catch(e) {
    clearTimeout(tid);
    throw e;
  }
}

async function fetchAllPrices() {
  const dot = document.getElementById("liveDot");
  const statusEl = document.getElementById("liveStatus");
  dot.className = "live-dot loading";
  statusEl.textContent = "Fetching live prices…";

  const tickerFunds = funds.filter(f => f.ticker);
  let fetched = 0, failed = 0;
  const errors = [];

  for (let i = 0; i < tickerFunds.length; i++) {
    const f = tickerFunds[i];
    try {
      const data = await fetchPrice(f.ticker);
      if (data) {
        funds = funds.map(x => x.id===f.id ? {...x, daily: data.changePct, fetched: true} : x);
        fetched++;
      } else {
        funds = funds.map(x => x.id===f.id ? {...x, daily: 0, fetched: true} : x);
        failed++;
        errors.push(`${f.ticker}: no data`);
      }
    } catch(e) {
      funds = funds.map(x => x.id===f.id ? {...x, daily: 0, fetched: true} : x);
      failed++;
      errors.push(`${f.ticker}: ${e.message}`);
    }

    pricesFetched = (fetched + failed) === tickerFunds.length;
    renderRows();
    updateDailyBadge();
    statusEl.textContent = `Fetching… ${fetched+failed}/${tickerFunds.length} funds`;
    if (i < tickerFunds.length-1) await new Promise(r => setTimeout(r, 200));
  }

  pricesFetched = true;
  dot.className = failed===tickerFunds.length ? "live-dot error" : "live-dot";
  const timeStr = new Date().toLocaleTimeString("en-US",{hour:"numeric",minute:"2-digit"});

  if (errors.length > 0) {
    console.group("Alpha Vantage fetch errors:");
    errors.forEach(e => console.warn(e));
    console.groupEnd();
    // Show first error in status bar to help diagnose
    statusEl.textContent = failed===0
      ? `All prices live · ${fetched} funds updated`
      : `${fetched} updated · ${failed} failed — check console (F12) · ${errors[0]}`;
  } else {
    statusEl.textContent = `All prices live · ${fetched} funds updated`;
  }
  document.getElementById("lastUpdated").textContent = `Last updated ${timeStr}`;
  renderRows();
  updateDailyBadge();
}

function updateProjection() {
  const balance=getBalance(), currentAge=parseInt(document.getElementById("currentAge").value)||55;
  const retireAge=parseInt(document.getElementById("retireAge").value)||65;
  const annualContrib=parseFloat(document.getElementById("annualContrib").value)||0;
  const r=(parseFloat(document.getElementById("growthRate").value)||5)/100;
  const years=Math.max(0,retireAge-currentAge);
  const labels=[],projected=[],conservative=[];
  for(let y=0;y<=years;y++){
    const base=balance*Math.pow(1+r,y);
    const contrib=r>0?annualContrib*((Math.pow(1+r,y)-1)/r):annualContrib*y;
    const rc=r*0.6;
    const cons=rc>0?balance*Math.pow(1+rc,y)+annualContrib*((Math.pow(1+rc,y)-1)/rc):balance+annualContrib*y;
    labels.push(currentAge+y); projected.push(Math.round(base+contrib)); conservative.push(Math.round(cons));
  }
  lineChart.data.labels=labels;
  lineChart.data.datasets[0].data=projected;
  lineChart.data.datasets[1].data=conservative;
  lineChart.update();
  document.getElementById("projFinal").textContent=fmt(projected[projected.length-1]||0);
  document.getElementById("psYears").textContent=`${years} yrs`;
}

  total===100 ? (el.textContent="✓ Fully allocated",el.style.color="#1B6B3A") : (el.textContent=`⚠ ${total.toFixed(1)}% allocated`,el.style.color="#B8470B");
}

function updateDailyBadge() {
  const balance=getBalance();
  const wDaily=funds.reduce((s,f)=>s+(f.percent/100)*f.daily,0);
  const dDollar=(wDaily/100)*balance;
  const badge=document.getElementById("dailyBadge");
  const text=document.getElementById("dailyText");
  if(!pricesFetched){ badge.className="badge badge-loading"; text.textContent="Fetching…"; return; }
  text.textContent=`${dDollar>=0?"+":""}${fmt(dDollar)} (${fmtPct(wDaily)})`;
  badge.className=`badge ${dDollar>=0?"badge-pos":"badge-neg"}`;
}

function addFund(){ funds.push({id:nextId++,name:"New Fund",percent:0,balance:0,gainLoss:0,ticker:null,daily:0}); renderAll(); }
